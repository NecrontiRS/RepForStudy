число положительных элементов каждого столбца матрицы.
Исходную матрицу вывести в общепринятом виде.
Результаты вывести в виде двух строк.

#include <iostream>
#include <string>

using namespace std;

int main()
{
    
    const int N = 3; //строки
    const int M = 5; //столбцы
    
    int sum;
    int num;
    
    int X[N][M] = 
    {
        { 1, -4, 2, 7, 3},
        { 4, 2, 5, 0, -1},
        { 2, 3, -6, -5, 2},
    };
    
     for(int i=0; i < N; i++) // перебор по строкам
    {
        cout << "["; // вывод скобки перед строкой
        
        for(int j=0; j < M; j++) // перебор по элементам в строке
        {
            cout << X[i][j] << "\t";
        }
        cout << "]" << endl; // вывод скобки в конце строки
    }
    
    for(int i=0; i < M; i++) //перебор по столбцам
    {
        
        for(int j=0; j < N; j++) //перебор по элементам в столбце
        {
            if ( X[j][i] >= 0 ){
                sum = sum + X[j][i];
                num = num + 1;
            }
        }
        cout << "Sum = " << sum << "; Num = " << num << endl;
        sum = 0;
        num = 0;
    }
}
